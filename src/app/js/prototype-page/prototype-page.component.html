Что такое prototype.

Prototype является прототипом всех объектов, и через него можно добавлять методы для работы с объектами 

  
    const person = {
    name: 'Max',
    age: 25,
    greet: () => {
        console.log('Greet!');
      }
    };
    // __proto__
    console.log(person.toString());

    const person2 = new Object({
    name: 'Miha',
    age: 28,
    greet: () => {
        console.log('Greet Miha!');
      }
    });
    Object.prototype.sayHello = () => {
    console.log('Say hello');
    };
    person2.sayHello();


    const lena = Object.create(person2);
    console.log(lena);
    // {}
    // __proto__ == persona2
    lena.name = 'Lena';
    console.log(lena);
    // {name: 'Lena'}
