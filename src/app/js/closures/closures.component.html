Замыкания ( Closures )

Замыкания - это функция внутри другой функции. 

  
    function createCalcFunc(n) {
      return function() {
        console.log(1000 * n)
      }
    }

    const calc = createCalcFunc(42)
    calc
    // несработает(возвращает функию но не вызывает)
    calc()
    // сработает, аозвращает и вызывает


    function createIncrement(n) {
      return function (num) {
        return n + num
      }
    }
    const addOne = createIncrement(1)
