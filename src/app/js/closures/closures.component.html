      function logPerson() {

    const person1 = {name: 'Mih', age: 22, job: 'Front'}
    const person2 = {name: 'Ln', age: 26, job: 'SMM'}

    bind(person1, logPerson)()
    bind(person2, logPerson)()

    function bind(person, fn){
      return function(...args){
        fn.apply(person, args)
      }
    }
