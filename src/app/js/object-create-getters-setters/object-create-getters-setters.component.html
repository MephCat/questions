      name: {
        value: 'NamePerson'
        enumerable: true,
        writable: true,
        configurable: true
      },
      birth: {
        value: 1993
        enumerable: false
      },
      age: {
        // getter - возвращает
        get () {
          return new Data().getFullYear() - this.birth
        },
        // принимает значение
        set (value) {
          console.log(value)
        }
      }
    })

    // configurable - позволяет удалять ключ из объекта
    delete person.name
    console.log(person)

    // writable - позволяет изменять поле в объекте
