Event Loop

Три хранилища
Call stack - то место где выполняется код(программа), по очерёдно закидывается для выполнения
Api - сюда складываются функции для ожидания выполнения (список отложеных функий) Callback Queue - цикл (тот самый event loop) делает итерацию по списку в очереди и закидывает в call stack latenflip.com 

  
    // Закидывает в стек и выполняет
    console.log('Start')
    // Закидывает в стек и выполняет
    console.log('Start 2')

    // Регистрирует функцию которая должна выполниться и ждёт когда будет готова к выполнению
    // после чего закидывает в стек для выполнения
    window.setTimeout( () => {
      console.log('Inside timeout')
    }, 1000)

    function to2sec(){
      console.log('after 2 sec')
    }
    setTimeout(to2sec, 2000)

    console.log('End')
    // отработает только после открытых консоль-логов из-за попадания в Callback Queue и API
    setTimeout(()=> console.log('insideTimeout 2'),0)
